      restrict: 'A',   // 'E', 'C', 'M'

Directive

  • Kompleks definisjon
  • Bør ikke ha en template
  • Brukes for å gi tilleggsfunksjonalitet til eksisterende DOM. F.eks:
    • ng-click
    • ng-repeat

Component

  • Kompakt definisjon
  • Må ha en template
  • Brukes for å bygge et custom DOM element

Struktur

Module

Module

  • Holder alt som har med modulen samlet på en plass.
  • Medfører enklere vedlikehold
  • Fremtvinger "Separation of concerns", gjennom naturlig fil-inndeling

index.js


// @ts-ignore
define([
  './route-name.module',
  './route-name.routes',
  './route-name.component'
], function () {
  'use strict'
});

*.module.js


define(['angular'], function (ng) {
  'use strict';

  /**
   * Define module dependencies inside [brackets]
   */
  return ng.module('route-name.module', []);
});

*.routes.js


define(['./route-name.module'], function (ngMod) {
  'use strict';

  /**
   * Define routing for this module
   */
  ngMod.config(['$stateProvider', function ($stateProvider) {
    $stateProvider
      .state('app.routename', {
        url   : '/routename',
        params: { },
        views : { 'content@app': 'routeNameComponent' },
        data  : { title: 'routename' }
      })
  }]);
});

*.component.js


define(['module', './route-name.module', 'css!./route-name.component.css'], function (module, ngMod) {
  'use strict';

  // Fetch the path of current file, so we can refer to template urls relative to this
  const path = module.uri.substr(0, module.uri.lastIndexOf('/'));

  /**
   * Component definition block
   */
  ngMod.component('routeNameComponent', {
    templateUrl : path + '/route-name.component.html',
    controllerAs: '$ctrl', // Not really nescessary - default is '$ctrl'
    controller  : RouteNameController,
    bindings    : {
      // Input
      'attribute': '<', // One-way binding into component

      // Output
      'onEvent'  : '&'  // Event fireing from component
    }
  });
        

  /**
   * Component controller block
   */
  RouteNameController.$inject = []; // Dependency injection
  function RouteNameController() {
    const $ctrl = this;

    /**
     * LIFECYCLE: Run on component intitialization.
     * This is basically the constructor
     */
    $ctrl.$onInit = function () {
    }

    /**
     * LIFECYCLE: Run on component tear down.
     * This is responsible for cleaning up the component
     */
    $ctrl.$onDestroy = function () {
    }
  }
});

Best practice

  • Unngå å bruke $scope
  • Aldri hent properties eller kall metoder via $scope.$parent
  • En controller holder state for en template.
    • Data manipulasjon eller kalkuleringer bør delegeres til services.
  • Inline styles gjør din template uleselig
  • Vær sparsommelig med biblioteker/plugins
    • De tar plass, som igjen går ut over ytelse og nedlastingstid.
    • De fleste plugins og bibliotek er for generisk/bloated for ditt spesifikke bruk. Hvis det ikke er for komplekst, og kan bygges selv innenfor rimelig tid, gjør det!
    • Unngå plugins som tvinger deg til å legge til enda et rammeverk.
  • Unngå å blande forskjellige rammeverk
    • Ikke miks angular/jquery/react/vue/osv... Bestem deg for et og bruk det.
    • Mye funksjonalitet ligger allerede til rette i web-api's.
      Du trenger f.eks ikke lodash for å filtrere eller manipulere en array: 
      arr
    .filter(item => item.isVisible == true)
    .map(item => ({name: item.firstName + item.lastName}));
  • Aldri repeter deg selv!
    • Hvis du dupliserer kode for å løse et nytt problem, har du feil løsning.
  • Refactoring er en konstant prosess
    • Løsninger du trodde var spesifikke når du skrev dem, kan vise seg å bli generiske når du ser mønstre utvikle seg.
    • Utsettes refactoring for lenge, kan du vikle deg inn "kode-floker" du vanskelig kommer ut av.
